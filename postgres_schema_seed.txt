-- PostgreSQL schema + sample seed data based on lib/db/db.js
-- Uses IF NOT EXISTS + ON CONFLICT DO NOTHING for safe re-execution.

BEGIN;

CREATE TABLE IF NOT EXISTS users (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    email TEXT UNIQUE,
    full_name TEXT,
    password TEXT,
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    created_at INTEGER DEFAULT (EXTRACT(EPOCH FROM NOW())::INTEGER)
);

CREATE TABLE IF NOT EXISTS roles (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL UNIQUE,
    description TEXT
);

CREATE TABLE IF NOT EXISTS permissions (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL UNIQUE,
    description TEXT
);

CREATE TABLE IF NOT EXISTS sources (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT UNIQUE NOT NULL,
    type TEXT NOT NULL, -- mongodb | s3
    config TEXT, -- JSON information
    created_by BIGINT,
    created_at INTEGER DEFAULT (EXTRACT(EPOCH FROM NOW())::INTEGER),
    FOREIGN KEY (created_by) REFERENCES users(id)
);

CREATE TABLE IF NOT EXISTS destinations (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT UNIQUE NOT NULL,
    type TEXT NOT NULL, -- s3 | host | local-storage
    config TEXT, -- JSON
    created_by BIGINT,
    created_at INTEGER DEFAULT (EXTRACT(EPOCH FROM NOW())::INTEGER),
    FOREIGN KEY (created_by) REFERENCES users(id)
);

CREATE TABLE IF NOT EXISTS targets (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    type TEXT NOT NULL, -- app | database | object-replication
    source_id BIGINT,
    created_by BIGINT,
    FOREIGN KEY (created_by) REFERENCES users(id),
    FOREIGN KEY (source_id) REFERENCES sources(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS notification_providers (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL UNIQUE,
    type TEXT NOT NULL CHECK (type IN ('smtp', 'ses')),
    config TEXT NOT NULL,
    is_enable BOOLEAN NOT NULL DEFAULT TRUE,
    created_at INTEGER DEFAULT (EXTRACT(EPOCH FROM NOW())::INTEGER),
    created_by BIGINT,
    FOREIGN KEY (created_by) REFERENCES users(id)
);

CREATE TABLE IF NOT EXISTS settings (
    key TEXT PRIMARY KEY,
    value TEXT NOT NULL,
    updated_at INTEGER,
    created_by BIGINT,
    FOREIGN KEY (created_by) REFERENCES users(id)
);

CREATE TABLE IF NOT EXISTS jobs (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL,
    is_enable BOOLEAN NOT NULL DEFAULT TRUE,
    status TEXT NOT NULL, -- pending | running | success | failed | canceled
    target TEXT,
    target_id BIGINT,
    schedule_type TEXT NOT NULL, -- interval | cron | manual
    schedule_value TEXT, -- e.g. 3600 or cron schema
    created_by BIGINT NOT NULL,
    created_at INTEGER NOT NULL,
    updated_at INTEGER NOT NULL,
    use_encryption BOOLEAN DEFAULT TRUE,
    retention_days INTEGER,
    FOREIGN KEY (created_by) REFERENCES users(id),
    FOREIGN KEY (target_id) REFERENCES targets(id)
);

CREATE TABLE IF NOT EXISTS backups (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name TEXT NOT NULL,
    backup_uid TEXT NOT NULL,
    user_id BIGINT NOT NULL,
    is_encrypted BOOLEAN NOT NULL DEFAULT TRUE,
    is_synced BOOLEAN NOT NULL DEFAULT FALSE,
    status TEXT NOT NULL, -- completed | failed | archived
    last_synced INTEGER DEFAULT (EXTRACT(EPOCH FROM NOW())::INTEGER),
    modified_at INTEGER DEFAULT (EXTRACT(EPOCH FROM NOW())::INTEGER),
    created_at INTEGER DEFAULT (EXTRACT(EPOCH FROM NOW())::INTEGER),
    storage TEXT,
    storage_conf TEXT,
    job_id BIGINT,
    archived_date INTEGER,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (job_id) REFERENCES jobs(id)
);

CREATE TABLE IF NOT EXISTS backup_files (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    storage_path TEXT NOT NULL,
    backup_id BIGINT NOT NULL,
    size BIGINT NOT NULL,
    status TEXT NOT NULL, -- online | archived
    type TEXT DEFAULT 'file' CHECK (type IN ('file', 'folder')),
    prefix TEXT,
    destination_folder TEXT,
    user_id BIGINT,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (backup_id) REFERENCES backups(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS role_permissions (
    role_id BIGINT NOT NULL,
    permission_id BIGINT NOT NULL,
    PRIMARY KEY (role_id, permission_id),
    FOREIGN KEY (role_id) REFERENCES roles(id),
    FOREIGN KEY (permission_id) REFERENCES permissions(id)
);

CREATE TABLE IF NOT EXISTS user_roles (
    user_id BIGINT NOT NULL,
    role_id BIGINT NOT NULL,
    PRIMARY KEY (user_id, role_id),
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (role_id) REFERENCES roles(id)
);

CREATE TABLE IF NOT EXISTS job_schedule_state (
    job_id BIGINT PRIMARY KEY,
    next_run_at INTEGER NOT NULL,
    last_run_at INTEGER,
    FOREIGN KEY (job_id) REFERENCES jobs(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS job_runs (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    job_id BIGINT NOT NULL,
    status TEXT NOT NULL, -- pending | running | success | failed | canceled
    started_at INTEGER,
    finished_at INTEGER,
    error_code TEXT,
    error_message TEXT,
    created_at INTEGER NOT NULL,
    FOREIGN KEY (job_id) REFERENCES jobs(id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS target_destinations (
    target_id BIGINT NOT NULL,
    destination_id BIGINT NOT NULL,
    created_by BIGINT,
    PRIMARY KEY (target_id, destination_id),
    FOREIGN KEY (target_id) REFERENCES targets(id) ON DELETE CASCADE,
    FOREIGN KEY (destination_id) REFERENCES destinations(id) ON DELETE CASCADE,
    FOREIGN KEY (created_by) REFERENCES users(id)
);
